- 다른 사람의 풀이를 보고 기겁했습니다. 이런 생각은 어떻게 하는 걸까요?

```js
function productMatrix(A, B) {
  return A.map(function (row) {
    return row.map(function (_, i) {
      return row.reduce(function (sum, cell, j) {
        return sum + cell * B[j][i];
      }, 0);
    });
  });
}
```
